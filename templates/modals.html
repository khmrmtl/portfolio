MINI MUSICAL KEYBOARD

Below is a short demo of the project.

Category: Arduino Project 
                                  

int buttonDo = 13;
int buttonRe = 12;
int buttonMi = 11;
int buttonFa = 10;
int buttonSo = 7;
int buttonLa = 6;
int buttonTi = 5;
int buttonDo2 = 4;

void setup()
{
  pinMode(buttonDo, INPUT);
  pinMode(buttonRe, INPUT);
  pinMode(buttonMi, INPUT);
  pinMode(buttonFa, INPUT);
  pinMode(buttonSo, INPUT);
  pinMode(buttonLa, INPUT);
  pinMode(buttonTi, INPUT);
  pinMode(buttonDo2, INPUT);
  pinMode(3, OUTPUT);
  Serial.begin(9600);
}

void loop()
{

  int input1=digitalRead(buttonDo);
  int input2=digitalRead(buttonRe);
  int input3=digitalRead(buttonMi);
  int input4=digitalRead(buttonFa);
  int input5=digitalRead(buttonSo);
  int input6=digitalRead(buttonLa);
  int input7=digitalRead(buttonTi);
  int input8=digitalRead(buttonDo2);

  if(input1==HIGH){       //if 1st button is pressed, play
    tone(3,261);          // c note
  }
  if(input2==HIGH){       //if 2nd button is pressed, play
    tone(3,294);          // d note
  }
  if(input3==HIGH){       //if 3rd button is pressed, play
    tone(3,329);          // e note
  }
  if(input4==HIGH){       //if 4th button is pressed, play
    tone(3,349);          // f note
  }
  if(input5==HIGH){       //if 5th button is pressed, play
    tone(3,392);          // g note
  }
  if(input6==HIGH){       //if 6th button is pressed, play
    tone(3,440);          // a note
  }
  if(input7==HIGH){       //if 7th button is pressed, play
    tone(3,493);          // b note
  }
  if(input8==HIGH){       //if 8th button is pressed, play
    tone(3,523);          // C note
  }

  delay(500);
  noTone(3);
}
